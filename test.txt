#include <stdio.h>

#define MAX_SIZE 5 // Define the maximum size of the stack

int stack[MAX_SIZE]; // Array to store stack elements
int top = -1;        // Index of the top element, initialized to -1 for an empty stack

// Function to check if the stack is empty
int isEmpty() {
    return (top == -1);
}

// Function to check if the stack is full
int isFull() {
    return (top == MAX_SIZE - 1);
}

// Function to add an element to the top of the stack (push operation)
void push(int data) {
    if (isFull()) {
        printf("Stack Overflow: Cannot push element, stack is full.\n");
    } else {
        stack[++top] = data; // Increment top and insert the data
        printf("%d pushed onto stack\n", data);
    }
}

// Function to remove and return the top element from the stack (pop operation)
int pop() {
    if (isEmpty()) {
        printf("Stack Underflow: Cannot pop element, stack is empty.\n");
        return -1; // Return -1 to indicate an error
    } else {
        int data = stack[top];
        top--; // Decrement top
        return data;
    }
}

// Function to return the top element without removing it (peek operation)
int peek() {
    if (isEmpty()) {
        printf("Stack is empty.\n");
        return -1; // Return -1 to indicate an error
    } else {
        return stack[top];
    }
}

// Main function to demonstrate stack operations
int main() {
    push(10);
    push(20);
    push(30);
    push(40);
    push(50);
    push(60); // This will cause a "Stack Overflow" error

    printf("\nElement at top of the stack: %d\n", peek());

    printf("\nElements popped from stack: \n");
    while (!isEmpty()) {
        printf("%d\n", pop());
    }

    printf("\nStack full: %s\n", isFull() ? "true" : "false");
    printf("Stack empty: %s\n", isEmpty() ? "true" : "false");

    return 0;
}
